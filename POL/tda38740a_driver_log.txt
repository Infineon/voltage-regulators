Please provide vout-voltage-multiplier values from devicetree.
Example:

i2c {
    #address-cells = <1>;
    #size-cells = <0>;

	hwmon@40 {
		/* Specify the compatible device */
		compatible = "infineon,tda38740a";
		/* Registry address */
		reg = <0x40>;
		/* vout-voltage-multiplier values */
		vout-voltage-multiplier = <70 75>;
	};

	hwmon@4F {
		/* Specify the compatible device */
		compatible = "infineon,tda38725a";
		/* Registry address */
		reg = <0x4F>;
		/* vout-voltage-multiplier values */
		vout-voltage-multiplier = <80 60>;
	};
};
